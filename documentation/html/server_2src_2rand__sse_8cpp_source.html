86  static const unsigned int mult[4] = {214013, 17405, 214013, 69069};
87  _ALIGN(16)
88  static const unsigned int gadd[4] = {2531011, 10395331, 13737667, 1};
89  _ALIGN(16)
90  static const unsigned int mask[4] = {0xFFFFFFFF, 0, 0xFFFFFFFF, 0};
91  _ALIGN(16)

94  adder = _mm_load_si128((__m128i *)gadd);
95  multiplier = _mm_load_si128((__m128i *)mult);
96  mod_mask = _mm_load_si128((__m128i *)mask);
97  sra_mask = _mm_load_si128((__m128i *)masklo);
98  cur_seed_split = _mm_shuffle_epi32(cur_seed, _MM_SHUFFLE(2, 3, 0, 1));
99 
100  cur_seed = _mm_mul_epu32(cur_seed, multiplier);
101  multiplier = _mm_shuffle_epi32(multiplier, _MM_SHUFFLE(2, 3, 0, 1));
102  cur_seed_split = _mm_mul_epu32(cur_seed_split, multiplier);
103 
104  cur_seed = _mm_and_si128(cur_seed, mod_mask);
105  cur_seed_split = _mm_and_si128(cur_seed_split, mod_mask);
106  cur_seed_split = _mm_shuffle_epi32(cur_seed_split, _MM_SHUFFLE(2, 3, 0, 1));
107  cur_seed = _mm_or_si128(cur_seed, cur_seed_split);
108  cur_seed = _mm_add_epi32(cur_seed, adder);
109 
110 #ifdef COMPATABILITY

116 #endif
