139 void destroy_win(WINDOW *local_win)
140 {
141  /* box(local_win, ' ', ' '); : This won't produce the desired
142  * result of erasing the window. It will leave it's four corners
143  * and so an ugly remnant of window.
144  */
145  wborder(local_win, ' ', ' ', ' ', ' ', ' ', ' ', ' ', ' ');
146  /* The parameters taken are
147  * 1. win: the window on which to operate
148  * 2. ls: character to be used for the left side of the window
149  * 3. rs: character to be used for the right side of the window
150  * 4. ts: character to be used for the top side of the window
151  * 5. bs: character to be used for the bottom side of the window
152  * 6. tl: character to be used for the top left corner of the window
153  * 7. tr: character to be used for the top right corner of the window
154  * 8. bl: character to be used for the bottom left corner of the window
155  * 9. br: character to be used for the bottom right corner of the window
156  */
157  wrefresh(local_win);
158  delwin(local_win);
159 }

161 void TUI::move()
162 {
163  WINDOW *my_win;
164  int startx, starty, width, height;
165  int ch;
166 
167  //initscr(); /* Start curses mode */
168  cbreak(); /* Line buffering disabled, Pass on
169  * everty thing to me */
170  keypad(stdscr, TRUE); /* I need that nifty F1 */
171 
172  height = 3;
173  width = 10;
174  starty = (LINES - height) / 2; /* Calculating for a center placement */
175  startx = (COLS - width) / 2; /* of the window */
176  printw("Press F1 to exit");
177  refresh();
178  my_win = create_newwin(height, width, starty, startx);
179 
180  while ((ch = getch()) != KEY_F(1))
181  {
182  switch (ch)
183  {
184  case KEY_LEFT:
185  destroy_win(my_win);
186  my_win = create_newwin(height, width, starty, --startx);
187  break;
188  case KEY_RIGHT:
189  destroy_win(my_win);
190  my_win = create_newwin(height, width, starty, ++startx);
191  break;
192  case KEY_UP:
193  destroy_win(my_win);
194  my_win = create_newwin(height, width, --starty, startx);
195  break;
196  case KEY_DOWN:
197  destroy_win(my_win);
198  my_win = create_newwin(height, width, ++starty, startx);
199  break;
200  }
201  }
202 }

291  {

295  mvwprintw(main_menu_win, h / 2 + i, (w / 2) - (vec_menu_items[i].size() / 2), vec_menu_items[i].c_str());
296  wattroff(main_menu_win, COLOR_PAIR(BR::CODE::MENU::COLOR::BLACK_WHITE));

315  }

319  }

329  }

338  choice--;

344  choice = 0;
345  }
346  else
347  {
348  choice++;
349  }
350  break;
351  case KEY_F(1):
352  exit(0);
353  break;
354  case 10:

356  break;
357  case '\r':
358  fl_exit = true;
359  break;
360  default:
361  break;
362  }
363  }
364 
365  switch (choice)
366  {
367  case 0:
368  ret = BR::CODE::MENU::MAIN::NEW_GAME;
369  break;
370  case 1:
371  ret = BR::CODE::MENU::MAIN::CONTINUE;
372  break;

375  break;
376  case 3:
377  ret = BR::CODE::MENU::MAIN::SETTINGS;
378  break;
379  case 4:
380  ret = BR::CODE::MENU::MAIN::HELP;
381  break;
382  case 5:
383  ret = BR::CODE::MENU::MAIN::EXIT;
384  break;

386  break;
387  }
